Singly Linked List

What is a Singly Linked List ?

A singly linked list is a linear data structure consisting of a sequence of nodes. Each node contains a data value and a reference (or link) to the next node in the sequence. It provides efficient insertion and deletion operations at both the beginning and middle of the list.

Singly Linked List Explanation:

A singly linked list is a sequence of nodes where each node points to the next node in the sequence. In this animation:

  • Node 1 points to Node 2
  • Node 2 points to Node 3
  • Node 3 points to Node 4
  • Node 4 points to NULL

Basic Operations on Singly Linked Lists

  • Insertion: Adding a new node at the beginning, end, or a specific position in the list.
  • Deletion: Removing a node from the beginning, end, or a specific position in the list.
  • Traversal: Visiting each node in the list to perform operations or retrieve values.
  • Searching: Finding a node with a specific value.
  • Displaying: Printing the contents of the list.

1. Implementing a Singly Linked List

Description: Implement a singly linked list with basic operations such as insertion, deletion, traversal, and display.

C


#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node* next;
};

void insertAtBeginning(struct Node** head, int value) {
    struct Node* new_node = (struct Node*) malloc(sizeof(struct Node));
    new_node->data = value;
    new_node->next = *head;
    *head = new_node;
}

void insertAtEnd(struct Node** head, int value) {
    struct Node* new_node = (struct Node*) malloc(sizeof(struct Node));
    struct Node* last = *head;
    new_node->data = value;
    new_node->next = NULL;

    if (*head == NULL) {
        *head = new_node;
        return;
    }

    while (last->next != NULL) {
        last = last->next;
    }
    last->next = new_node;
}

void deleteNode(struct Node** head, int value) {
    struct Node* temp = *head;
    struct Node* prev = NULL;

    if (temp != NULL && temp->data == value) {
        *head = temp->next;
        free(temp);
        return;
    }

    while (temp != NULL && temp->data != value) {
        prev = temp;
        temp = temp->next;
    }

    if (temp == NULL) return;

    prev->next = temp->next;
    free(temp);
}

void traverseList(struct Node* head) {
    struct Node* temp = head;
    while (temp != NULL) {
        printf("%d -> ", temp->data);
        temp = temp->next;
    }
    printf("NULL\n");
}

int main() {
    struct Node* head = NULL;
    insertAtBeginning(&head, 10);
    insertAtBeginning(&head, 20);
    insertAtEnd(&head, 30);
    traverseList(head);
    deleteNode(&head, 20);
    traverseList(head);
    return 0;
}

C++


#include <iostream>

class Node {
public:
    int data;
    Node* next;
    Node(int value) : data(value), next(nullptr) {}
};

class SinglyLinkedList {
    Node* head;
public:
    SinglyLinkedList() : head(nullptr) {}

    void insertAtBeginning(int value) {
        Node* new_node = new Node(value);
        new_node->next = head;
        head = new_node;
    }

    void insertAtEnd(int value) {
        Node* new_node = new Node(value);
        if (head == nullptr) {
            head = new_node;
            return;
        }
        Node* last = head;
        while (last->next != nullptr) {
            last = last->next;
        }
        last->next = new_node;
    }

    void deleteNode(int value) {
        Node* temp = head;
        Node* prev = nullptr;

        if (temp != nullptr && temp->data == value) {
            head = temp->next;
            delete temp;
            return;
        }

        while (temp != nullptr && temp->data != value) {
            prev = temp;
            temp = temp->next;
        }

        if (temp == nullptr) return;

        prev->next = temp->next;
        delete temp;
    }

    void traverseList() {
        Node* temp = head;
        while (temp != nullptr) {
            std::cout << temp->data << " -> ";
            temp = temp->next;
        }
        std::cout << "NULL\n";
    }
};

int main() {
    SinglyLinkedList list;
    list.insertAtBeginning(10);
    list.insertAtBeginning(20);
    list.insertAtEnd(30);
    list.traverseList();
    list.deleteNode(20);
    list.traverseList();
    return 0;
}

            last = last.next;

class Node:
    def __init__(self, value):
        self.data = value
        self.next = None

class SinglyLinkedList:
    def __init__(self):
        self.head = None

    def insert_at_beginning(self, value):
        new_node = Node(value)
        new_node.next = self.head
        self.head = new_node

    def insert_at_end(self, value):
        new_node = Node(value)
        if self.head is None:
            self.head = new_node
            return
        last = self.head
        while last.next:
            last = last.next
        last.next = new_node

    def delete_node(self, value):
        temp = self.head
        prev = None

        if temp is not None and temp.data == value:
            self.head = temp.next
            return

        while temp is not None and temp.data != value:
            prev = temp
            temp = temp.next

        if temp is None:
            return

        prev.next = temp.next

    def traverse_list(self):
        temp = self.head
        while temp:
            print(temp.data, end=" -> ")
            temp = temp.next
        print("NULL")

Real-World Problems and Applications

  • Dynamic Memory Allocation: Singly linked lists are useful in situations where the number of elements is not known in advance and dynamic memory allocation is required.
  • Queue Implementation: Singly linked lists can be used to implement queue data structures, allowing efficient enqueue and dequeue operations.
  • Polynomial Representation: Polynomials can be represented and manipulated using singly linked lists, where each node represents a term in the polynomial.
  • Undo Functionality: In applications that require undo functionality, singly linked lists can store previous states of the application to allow users to revert to previous actions.

Summary

Singly linked lists are a fundamental data structure that provide efficient ways to store and manipulate a sequence of elements. With basic operations like insertion, deletion, and traversal, singly linked lists are versatile and useful for various real-world applications.
