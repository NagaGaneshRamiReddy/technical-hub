Dijkstra's Algorithm

Introduction

Dijkstra's Algorithm is a famous shortest path algorithm that finds the shortest path between nodes in a graph. It is particularly useful for graphs with non-negative weights. The algorithm works by iteratively selecting the node with the smallest distance and updating its neighbors.

How It Works

  1. Initialize distances from the source node to all other nodes as infinity, and the distance to itself as 0.
  2. Use a priority queue to select the node with the smallest distance.
  3. Update the distances to neighboring nodes.
  4. Repeat the process until all nodes have been processed.

Implementation in C++


#include <iostream>
#include <vector>
#include <queue>

const int INF = 1e9;

class Graph {
    int V;
    std::vector>> adjList;

public:
    Graph(int V) : V(V), adjList(V) {}

    void addEdge(int u, int v, int w) {
        adjList[u].push_back({v, w});
        adjList[v].push_back({u, w}); // For undirected graph
    }

    void dijkstra(int start) {
        std::vector dist(V, INF);
        dist[start] = 0;
        std::priority_queue, std::vector>, std::greater<>> pq;
        pq.push({0, start});

        while (!pq.empty()) {
            int u = pq.top().second;
            pq.pop();

            for (const auto& neighbor : adjList[u]) {
                int v = neighbor.first;
                int weight = neighbor.second;

                if (dist[u] + weight < dist[v]) {
                    dist[v] = dist[u] + weight;
                    pq.push({dist[v], v});
                }
            }
        }

        for (int i = 0; i < V; ++i) {
            std::cout << "Distance from " << start << " to " << i << " is " << (dist[i] == INF ? "INF" : std::to_string(dist[i])) << std::endl;
        }
    }
};

int main() {
    Graph g(5);
    g.addEdge(0, 1, 10);
    g.addEdge(0, 4, 3);
    g.addEdge(1, 2, 2);
    g.addEdge(1, 4, 4);
    g.addEdge(2, 3, 9);
    g.addEdge(3, 4, 7);
    g.dijkstra(0);
    return 0;
}

Implementation in Python


import heapq

class Graph:
    def __init__(self, vertices):
        self.V = vertices
        self.adjList = [[] for _ in range(vertices)]

    def add_edge(self, u, v, w):
        self.adjList[u].append((v, w))
        self.adjList[v].append((u, w))  # For undirected graph

    def dijkstra(self, start):
        dist = [float('inf')] * self.V
        dist[start] = 0
        pq = [(0, start)]

        while pq:
            d, u = heapq.heappop(pq)

            if d > dist[u]:
                continue

            for v, weight in self.adjList[u]:
                if dist[u] + weight < dist[v]:
                    dist[v] = dist[u] + weight
                    heapq.heappush(pq, (dist[v], v))

        for i in range(self.V):
            print(f"Distance from {start} to {i} is {'INF' if dist[i] == float('inf') else dist[i]}")

# Example usage
g = Graph(5)
g.add_edge(0, 1, 10)
g.add_edge(0, 4, 3)
g.add_edge(1, 2, 2)
g.add_edge(1, 4, 4)
g.add_edge(2, 3, 9)
g.add_edge(3, 4, 7)
g.dijkstra(0)

Implementation in Java


import java.util.*;

class Graph {
    private final int V;
    private final List> adjList;

    static class Pair {
        int vertex, weight;
        Pair(int v, int w) {
            vertex = v;
            weight = w;
        }
    }

    Graph(int V) {
        this.V = V;
        adjList = new ArrayList<>();
        for (int i = 0; i < V; i++) {
            adjList.add(new ArrayList<>());
        }
    }

    void addEdge(int u, int v, int w) {
        adjList.get(u).add(new Pair(v, w));
        adjList.get(v).add(new Pair(u, w)); // For undirected graph
    }

    void dijkstra(int start) {
        int[] dist = new int[V];
        Arrays.fill(dist, Integer.MAX_VALUE);
        dist[start] = 0;
        PriorityQueue pq = new PriorityQueue<>(Comparator.comparingInt(pair -> pair.weight));
        pq.add(new Pair(start, 0));

        while (!pq.isEmpty()) {
            int u = pq.poll().vertex;

            for (Pair neighbor : adjList.get(u)) {
                int v = neighbor.vertex;
                int weight = neighbor.weight;

                if (dist[u] + weight < dist[v]) {
                    dist[v] = dist[u] + weight;
                    pq.add(new Pair(v, dist[v]));
                }
            }
        }

        for (int i = 0; i < V; i++) {
            System.out.println("Distance from " + start + " to " + i + " is " + (dist[i] == Integer.MAX_VALUE ? "INF" : dist[i]));
        }
    }
}

public class Main {
    public static void main(String[] args) {
        Graph g = new Graph(5);
        g.addEdge(0, 1, 10);
        g.addEdge(0, 4, 3);
        g.addEdge(1, 2, 2);
        g.addEdge(1, 4, 4);
        g.addEdge(2, 3, 9);
        g.addEdge(3, 4, 7);
        g.dijkstra(0);
    }
}

Summary

Dijkstra's Algorithm is a powerful method for finding the shortest paths from a source node to all other nodes in a graph with non-negative weights. It efficiently handles graphs using a priority queue to explore the shortest path first. This algorithm is widely used in network routing, geographical mapping, and other fields requiring optimal pathfinding.
