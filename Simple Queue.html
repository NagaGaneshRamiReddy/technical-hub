Simple Queue

What is a Simple Queue ?

A simple queue is a linear data structure that follows the First In, First Out (FIFO) principle. In this queue, the first element added to the queue will be the first one to be removed. This is akin to people lining up at a checkout counter: the first person in line is the first to be served.

Queue Explanation:

A queue is a data structure that follows the First In, First Out (FIFO) principle. The operations are:

  • Enqueue: Adds an item to the end of the queue.
  • Dequeue: Removes the item from the front of the queue.

Basic Operations on Simple Queues

  • Enqueue: Adds an element to the end of the queue.
  • Dequeue: Removes an element from the front of the queue.
  • Peek/Front: Retrieves the element at the front of the queue without removing it.
  • IsEmpty: Checks if the queue is empty.
  • Size: Returns the number of elements in the queue.

1. Implementing a Simple Queue

Description: Implement a simple queue with basic operations such as enqueue, dequeue, peek, and check if the queue is empty. This implementation will help you understand the core principles of managing a FIFO data structure.

C


#include <stdio.h>
#include <stdlib.h>

#define MAX 100

typedef struct {
    int front, rear, size;
    int array[MAX];
} Queue;

void initQueue(Queue* q) {
    q->front = 0;
    q->rear = -1;
    q->size = 0;
}

int isEmpty(Queue* q) {
    return q->size == 0;
}

int isFull(Queue* q) {
    return q->size == MAX;
}

void enqueue(Queue* q, int value) {
    if (isFull(q)) {
        printf("Queue is full\n");
        return;
    }
    q->rear = (q->rear + 1) % MAX;
    q->array[q->rear] = value;
    q->size++;
}

int dequeue(Queue* q) {
    if (isEmpty(q)) {
        printf("Queue is empty\n");
        return -1;
    }
    int value = q->array[q->front];
    q->front = (q->front + 1) % MAX;
    q->size--;
    return value;
}

int peek(Queue* q) {
    if (isEmpty(q)) {
        printf("Queue is empty\n");
        return -1;
    }
    return q->array[q->front];
}

int main() {
    Queue q;
    initQueue(&q);
    enqueue(&q, 10);
    enqueue(&q, 20);
    printf("Front element: %d\n", peek(&q));
    printf("Dequeued: %d\n", dequeue(&q));
    printf("Front element: %d\n", peek(&q));
    return 0;
}

C++


#include <iostream>>
#include <deque>>

class SimpleQueue {
    std::deque queue;
public:
    void enqueue(int value) {
        queue.push_back(value);
    }

    int dequeue() {
        if (queue.empty()) {
            std::cerr << "Queue is empty" << std::endl;
            return -1;
        }
        int value = queue.front();
        queue.pop_front();
        return value;
    }

    int peek() {
        if (queue.empty()) {
            std::cerr << "Queue is empty" << std::endl;
            return -1;
        }
        return queue.front();
    }

    bool isEmpty() {
        return queue.empty();
    }
};

int main() {
    SimpleQueue q;
    q.enqueue(10);
    q.enqueue(20);
    std::cout << "Front element: " << q.peek() << std::endl;
    std::cout << "Dequeued: " << q.dequeue() << std::endl;
    std::cout << "Front element: " << q.peek() << std::endl;
    return 0;
}

Java


import java.util.LinkedList;
import java.util.Queue;

class SimpleQueue {
    private Queue queue;

    SimpleQueue() {
        queue = new LinkedList<>();
    }

    void enqueue(int value) {
        queue.add(value);
    }

    int dequeue() {
        if (queue.isEmpty()) {
            System.out.println("Queue is empty");
            return -1;
        }
        return queue.poll();
    }

    int peek() {
        if (queue.isEmpty()) {
            System.out.println("Queue is empty");
            return -1;
        }
        return queue.peek();
    }

    boolean isEmpty() {
        return queue.isEmpty();
    }

    public static void main(String[] args) {
        SimpleQueue q = new SimpleQueue();
        q.enqueue(10);
        q.enqueue(20);
        System.out.println("Front element: " + q.peek());
        System.out.println("Dequeued: " + q.dequeue());
        System.out.println("Front element: " + q.peek());
    }
}

Python


from collections import deque

class SimpleQueue:
    def __init__(self):
        self.queue = deque()

    def enqueue(self, value):
        self.queue.append(value)

    def dequeue(self):
        if not self.queue:
            print("Queue is empty")
            return -1
        return self.queue.popleft()

    def peek(self):
        if not self.queue:
            print("Queue is empty")
            return -1
        return self.queue[0]

    def is_empty(self):
        return len(self.queue) == 0

# Example usage
q = SimpleQueue()
q.enqueue(10)
q.enqueue(20)
print("Front element:", q.peek())
print("Dequeued:", q.dequeue())
print("Front element:", q.peek())
