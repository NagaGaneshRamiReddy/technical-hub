Priority Queue

What is a Priority Queue ?

A priority queue is a type of queue where each element has a priority associated with it. In a priority queue, elements are dequeued based on their priority rather than their insertion order. Elements with higher priority are processed before elements with lower priority. Priority queues are commonly implemented using heaps, but they can also be implemented with other data structures like ordered lists or binary search trees.

Priority Queue Explanation:

A priority queue is a data structure where each element has a priority. Elements with higher priority are served before those with lower priority. The operations are:

  • Enqueue: Adds an item to the queue with a specified priority. Items are placed in the queue based on their priority.
  • Dequeue: Removes the item with the highest priority from the queue.

Basic Operations on Priority Queues

  • Insert: Adds an element to the priority queue with an associated priority.
  • Extract-Max/Min: Removes and returns the element with the highest (or lowest) priority.
  • Peek: Retrieves the element with the highest (or lowest) priority without removing it.
  • Increase/Decrease Priority: Updates the priority of an existing element in the queue.

1. Implementing a Priority Queue

Description: Implement a priority queue with basic operations such as insertion, extraction of the maximum (or minimum), and priority updates. This implementation demonstrates how a priority queue can be managed using a heap data structure.

C (Using Max-Heap)


#include <stdio.h>
#include <stdlib.h>

#define MAX 100

typedef struct {
    int data;
    int priority;
} Element;

typedef struct {
    Element heap[MAX];
    int size;
} PriorityQueue;

void swap(Element* a, Element* b) {
    Element temp = *a;
    *a = *b;
    *b = temp;
}

void heapify(PriorityQueue* pq, int index) {
    int largest = index;
    int left = 2 * index + 1;
    int right = 2 * index + 2;

    if (left < pq->size && pq->heap[left].priority > pq->heap[largest].priority) {
        largest = left;
    }
    if (right < pq->size && pq->heap[right].priority > pq->heap[largest].priority) {
        largest = right;
    }
    if (largest != index) {
        swap(&pq->heap[index], &pq->heap[largest]);
        heapify(pq, largest);
    }
}

void insert(PriorityQueue* pq, int data, int priority) {
    if (pq->size == MAX) {
        printf("Queue is full\n");
        return;
    }
    pq->heap[pq->size].data = data;
    pq->heap[pq->size].priority = priority;
    pq->size++;
    for (int i = pq->size / 2 - 1; i >= 0; i--) {
        heapify(pq, i);
    }
}

Element extractMax(PriorityQueue* pq) {
    if (pq->size == 0) {
        printf("Queue is empty\n");
        Element e = {-1, -1};
        return e;
    }
    Element root = pq->heap[0];
    pq->heap[0] = pq->heap[pq->size - 1];
    pq->size--;
    heapify(pq, 0);
    return root;
}

int main() {
    PriorityQueue pq;
    pq.size = 0;

    insert(&pq, 10, 1);
    insert(&pq, 20, 2);
    insert(&pq, 30, 3);

    Element max = extractMax(&pq);
    printf("Extracted Max: %d with priority %d\n", max.data, max.priority);

    return 0;
}

C++ (Using Max-Heap)


#include <iostream>>
#include <queue>>
#include <vector>>

class PriorityQueue {
    std::priority_queue> pq; // Pair of (priority, data)
public:
    void insert(int data, int priority) {
        pq.push(std::make_pair(priority, data));
    }

    int extractMax() {
        if (pq.empty()) {
            std::cerr << "Queue is empty" << std::endl;
            return -1;
        }
        int data = pq.top().second;
        pq.pop();
        return data;
    }

    int peek() {
        if (pq.empty()) {
            std::cerr << "Queue is empty" << std::endl;
            return -1;
        }
        return pq.top().second;
    }
};

int main() {
    PriorityQueue pq;
    pq.insert(10, 1);
    pq.insert(20, 2);
    pq.insert(30, 3);

    std::cout << "Extracted Max: " << pq.extractMax() << std::endl;

    return 0;
}

Java (Using PriorityQueue Class)


import java.util.PriorityQueue;

class Element implements Comparable {
    int data;
    int priority;

    Element(int data, int priority) {
        this.data = data;
        this.priority = priority;
    }

    @Override
    public int compareTo(Element other) {
        return Integer.compare(other.priority, this.priority);
    }
}

class PriorityQueueExample {
    private PriorityQueue pq = new PriorityQueue<>();

    void insert(int data, int priority) {
        pq.offer(new Element(data, priority));
    }

    int extractMax() {
        if (pq.isEmpty()) {
            System.out.println("Queue is empty");
            return -1;
        }
        return pq.poll().data;
    }

    int peek() {
        if (pq.isEmpty()) {
            System.out.println("Queue is empty");
            return -1;
        }
        return pq.peek().data;
    }

    public static void main(String[] args) {
        PriorityQueueExample pq = new PriorityQueueExample();
        pq.insert(10, 1);
        pq.insert(20, 2);
        pq.insert(30, 3);

        System.out.println("Extracted Max: " + pq.extractMax());
    }
}

Python (Using heapq for Min-Heap)


import heapq

class PriorityQueue:
    def __init__(self):
        self.heap = []

    def insert(self, data, priority):
        heapq.heappush(self.heap, (-priority, data))  # Python heapq is a min-heap, so we use negative priority for max-heap behavior

    def extract_max(self):
        if not self.heap:
            print("Queue is empty")
            return None
        return heapq.heappop(self.heap)[1]

    def peek(self):
        if not self.heap:
            print("Queue is empty")
            return None
        return self.heap[0][1]

# Example usage
pq = PriorityQueue()
pq.insert(10, 1)
pq.insert(20, 2)
pq.insert(30, 3)

print("Extracted Max:", pq.extract_max())

Summary

Priority queues are specialized data structures where each element is associated with a priority. They allow for efficient retrieval and removal of elements based on their priority rather than their order of insertion. Priority queues are widely used in applications such as scheduling tasks, managing events, and implementing algorithms like Dijkstra's shortest path.
