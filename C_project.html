Calculator

The calculator project is a fundamental programming exercise that allows you to perform basic arithmetic operations like addition, subtraction, multiplication, and division. It serves as an excellent introduction to the use of variables, conditional statements, and user input in C programming.

This project uses a switch statement to determine which operation to execute based on the user-provided operator (+, -, *, /). Input validation is an essential part of the program, especially for division. Attempting to divide by zero is mathematically undefined and results in an error. This is handled gracefully in the code, making it a valuable lesson in defensive programming.

Tip: Remember to handle invalid input, like dividing by zero!

#include <stdio.h>

int main() {
    double num1, num2, result;
    char operator;

    printf("Enter first number: ");
    scanf("%lf", &num1);

    printf("Enter an operator (+, -, *, /): ");
    scanf(" %c", &operator);

    printf("Enter second number: ");
    scanf("%lf", &num2);

    switch (operator) {
        case '+': result = num1 + num2; break;
        case '-': result = num1 - num2; break;
        case '*': result = num1 * num2; break;
        case '/': 
            if (num2 != 0)
                result = num1 / num2; 
            else {
                printf("Error! Division by zero.\n");
                return 1;
            }
            break;
        default:
            printf("Invalid operator!\n");
            return 1;
    }

    printf("Result: %.2lf\n", result);
    return 0;
}

Number Guessing Game

The number guessing game is an interactive and engaging project that introduces you to the concepts of loops, conditionals, and random number generation. The program generates a random number between 1 and 100 using the rand() function, seeded by srand(time(0)) to ensure randomness every time the program runs.

The user inputs guesses, and the program provides feedback such as "Too high!" or "Too low!" until the correct number is guessed. It also keeps track of the number of attempts, offering a fun way to learn about do-while loops. Enhancing the game can include adding difficulty levels or limiting the number of attempts.

Tip: Use srand(time(0)); to seed the random number generator.

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    int number, guess, attempts = 0;
    srand(time(0));

    number = rand() % 100 + 1;

    printf("Guess the number between 1 and 100: ");

    do {
        scanf("%d", &guess);
        attempts++;

        if (guess < number) {
            printf("Too low! Try again: ");
        } else if (guess > number) {
            printf("Too high! Try again: ");
        } else {
            printf("Correct! You guessed the number in %d attempts.\n", attempts);
        }
    } while (guess != number);

    return 0;
}

Library Management System

A system to manage library books, including issuing and returning books.

Tip: Use arrays or file storage to maintain the database of books.

        #include <stdio.h>
        #include <string.h>
        
        #define MAX_BOOKS 100
        
        typedef struct {
            char title[50];
            char author[50];
            int available;
        } Book;
        
        Book library[MAX_BOOKS];
        int totalBooks = 0;
        
        void addBook() {
            if (totalBooks >= MAX_BOOKS) {
                printf("Library is full!\n");
                return;
            }
            printf("Enter book title: ");
            scanf(" %[^\n]", library[totalBooks].title);
            printf("Enter author: ");
            scanf(" %[^\n]", library[totalBooks].author);
            library[totalBooks].available = 1;
            totalBooks++;
            printf("Book added successfully!\n");
        }
        
        void issueBook() {
            char title[50];
            printf("Enter book title to issue: ");
            scanf(" %[^\n]", title);
            for (int i = 0; i < totalBooks; i++) {
                if (strcmp(library[i].title, title) == 0 && library[i].available) {
                    library[i].available = 0;
                    printf("Book issued successfully!\n");
                    return;
                }
            }
            printf("Book not available or not found!\n");
        }
        
        void returnBook() {
            char title[50];
            printf("Enter book title to return: ");
            scanf(" %[^\n]", title);
            for (int i = 0; i < totalBooks; i++) {
                if (strcmp(library[i].title, title) == 0 && !library[i].available) {
                    library[i].available = 1;
                    printf("Book returned successfully!\n");
                    return;
                }
            }
            printf("Book not found!\n");
        }
        
        void viewBooks() {
            printf("Library Books:\n");
            for (int i = 0; i < totalBooks; i++) {
                printf("Title: %s, Author: %s, Available: %s\n", library[i].title, library[i].author, library[i].available ? "Yes" : "No");
            }
        }
        
        int main() {
            int choice;
            do {
                printf("\nLibrary Management System\n");
                printf("1. Add Book\n2. Issue Book\n3. Return Book\n4. View Books\n5. Exit\n");
                printf("Enter your choice: ");
                scanf("%d", &choice);
        
                switch (choice) {
                    case 1: addBook(); break;
                    case 2: issueBook(); break;
                    case 3: returnBook(); break;
                    case 4: viewBooks(); break;
                    case 5: printf("Exiting...\n"); break;
                    default: printf("Invalid choice!\n");
                }
            } while (choice != 5);
            return 0;
        }

Tic-Tac-Toe Game

A two-player game of Tic-Tac-Toe.

Tip: Use a 3x3 array to represent the game board.

        #include <stdio.h>
        
        char board[3][3];
        int currentPlayer;
        
        void initBoard() {
            for (int i = 0; i < 3; i++)
                for (int j = 0; j < 3; j++)
                    board[i][j] = ' ';
        }
        
        void displayBoard() {
            printf("\n");
            for (int i = 0; i < 3; i++) {
                for (int j = 0; j < 3; j++) {
                    printf(" %c ", board[i][j]);
                    if (j < 2) printf("|");
                }
                if (i < 2) printf("\n---|---|---\n");
            }
            printf("\n");
        }
        
        int checkWin() {
            for (int i = 0; i < 3; i++) {
                if (board[i][0] == board[i][1] && board[i][1] == board[i][2] && board[i][0] != ' ')
                    return 1;
                if (board[0][i] == board[1][i] && board[1][i] == board[2][i] && board[0][i] != ' ')
                    return 1;
            }
            if (board[0][0] == board[1][1] && board[1][1] == board[2][2] && board[0][0] != ' ')
                return 1;
            if (board[0][2] == board[1][1] && board[1][1] == board[2][0] && board[0][2] != ' ')
                return 1;
            return 0;
        }
        
        int checkDraw() {
            for (int i = 0; i < 3; i++)
                for (int j = 0; j < 3; j++)
                    if (board[i][j] == ' ')
                        return 0;
            return 1;
        }
        
        void playGame() {
            int row, col;
            char mark = (currentPlayer == 1) ? 'X' : 'O';
        
            printf("Player %d (%c), enter row and column (1-3): ", currentPlayer, mark);
            scanf("%d %d", &row, &col);
            row--; col--;
        
            if (row >= 0 && row < 3 && col >= 0 && col < 3 && board[row][col] == ' ') {
                board[row][col] = mark;
                if (checkWin()) {
                    displayBoard();
                    printf("Player %d wins!\n", currentPlayer);
                    return;
                }
                if (checkDraw()) {
                    displayBoard();
                    printf("It's a draw!\n");
                    return;
                }
                currentPlayer = (currentPlayer == 1) ? 2 : 1;
            } else {
                printf("Invalid move. Try again.\n");
            }
            displayBoard();
            playGame();
        }
        
        int main() {
            printf("Tic-Tac-Toe Game\n");
            initBoard();
            currentPlayer = 1;
            displayBoard();
            playGame();
            return 0;
        }

Sorting Algorithm Visualizer

Visualize sorting algorithms like Bubble Sort and Quick Sort in the console.

Tip: Use a delay function to simulate the sorting process visually.

        #include <stdio.h>
        
        void displayArray(int arr[], int size) {
            for (int i = 0; i < size; i++) {
                printf("%d ", arr[i]);
            }
            printf("\n");
        }
        
        void bubbleSort(int arr[], int size) {
            for (int i = 0; i < size - 1; i++) {
                for (int j = 0; j < size - i - 1; j++) {
                    if (arr[j] > arr[j + 1]) {
                        int temp = arr[j];
                        arr[j] = arr[j + 1];
                        arr[j + 1] = temp;
                        displayArray(arr, size);
                    }
                }
            }
        }
        
        int main() {
            int arr[] = {5, 3, 8, 6, 2};
            int size = sizeof(arr) / sizeof(arr[0]);
        
            printf("Original Array:\n");
            displayArray(arr, size);
        
            printf("\nBubble Sort Visualization:\n");
            bubbleSort(arr, size);
        
            printf("\nSorted Array:\n");
            displayArray(arr, size);
            return 0;
        }

Mini Shell

Implement a command-line shell that executes basic terminal commands like ls, cd, or echo.

Tip: Use the fork() and exec() system calls for process management.

        #include <stdio.h>
        #include <stdlib.h>
        #include <unistd.h>
        #include <string.h>
        
        #define MAX_INPUT 1024
        #define MAX_ARGS 64
        
        void executeCommand(char *input) {
            char *args[MAX_ARGS];
            char *token = strtok(input, " ");
            int i = 0;
        
            while (token != NULL && i < MAX_ARGS - 1) {
                args[i++] = token;
                token = strtok(NULL, " ");
            }
            args[i] = NULL;
        
            if (fork() == 0) {
                execvp(args[0], args);
                perror("Error");
                exit(EXIT_FAILURE);
            } else {
                wait(NULL);
            }
        }
        
        int main() {
            char input[MAX_INPUT];
        
            while (1) {
                printf("mini-shell> ");
                fgets(input, MAX_INPUT, stdin);
                input[strcspn(input, "\n")] = 0; // Remove trailing newline
        
                if (strcmp(input, "exit") == 0) break;
                executeCommand(input);
            }
        
            printf("Exiting shell...\n");
            return 0;
        }

Compiler Design

Build a mini compiler that parses and evaluates arithmetic expressions.

Tip: Use recursive descent parsing for evaluating arithmetic expressions.

        #include <stdio.h>
        #include <ctype.h>
        
        char *expr;
        int parseExpression();
        
        int parseNumber() {
            int num = 0;
            while (isdigit(*expr)) {
                num = num * 10 + (*expr - '0');
                expr++;
            }
            return num;
        }
        
        int parseFactor() {
            if (*expr == '(') {
                expr++;
                int result = parseExpression();
                if (*expr == ')') expr++;
                return result;
            } else {
                return parseNumber();
            }
        }
        
        int parseTerm() {
            int result = parseFactor();
            while (*expr == '*' || *expr == '/') {
                char op = *expr++;
                if (op == '*')
                    result *= parseFactor();
                else
                    result /= parseFactor();
            }
            return result;
        }
        
        int parseExpression() {
            int result = parseTerm();
            while (*expr == '+' || *expr == '-') {
                char op = *expr++;
                if (op == '+')
                    result += parseTerm();
                else
                    result -= parseTerm();
            }
            return result;
        }
        
        int main() {
            char input[100];
            printf("Enter expression: ");
            scanf("%s", input);
            expr = input;
            int result = parseExpression();
            printf("Result: %d\n", result);
            return 0;
        }

Chat Application

Create a simple chat application using socket programming to send and receive messages.

Tip: Use separate threads for sending and receiving messages simultaneously.

        #include <stdio.h>
        #include <stdlib.h>
        #include <string.h>
        #include <unistd.h>
        #include <arpa/inet.h>
        #include <pthread.h>
        
        #define PORT 8080
        #define MAXLINE 1024
        
        void *sendMessage(void *sock) {
            int sockfd = *(int *)sock;
            char buffer[MAXLINE];
        
            while (1) {
                fgets(buffer, MAXLINE, stdin);
                send(sockfd, buffer, strlen(buffer), 0);
            }
            return NULL;
        }
        
        void *receiveMessage(void *sock) {
            int sockfd = *(int *)sock;
            char buffer[MAXLINE];
            while (1) {
                int n = recv(sockfd, buffer, MAXLINE, 0);
                buffer[n] = '\0';
                printf("Server: %s\n", buffer);
            }
            return NULL;
        }
        
        int main() {
            int sockfd;
            struct sockaddr_in servaddr;
            pthread_t sendThread, recvThread;
        
            sockfd = socket(AF_INET, SOCK_STREAM, 0);
            servaddr.sin_family = AF_INET;
            servaddr.sin_port = htons(PORT);
            inet_pton(AF_INET, "127.0.0.1", &servaddr.sin_addr);
        
            connect(sockfd, (struct sockaddr *)&servaddr, sizeof(servaddr));
            printf("Connected to the server.\n");
        
            pthread_create(&sendThread, NULL, sendMessage, &sockfd);
            pthread_create(&recvThread, NULL, receiveMessage, &sockfd);
        
            pthread_join(sendThread, NULL);
            pthread_join(recvThread, NULL);
            close(sockfd);
            return 0;
        }

Sudoku Solver

Develop a program that takes a Sudoku puzzle as input and solves it using backtracking.

Tip: Use recursion for backtracking to try all possible values for each cell.

        #include <stdio.h>
        
        #define N 9
        
        int isSafe(int grid[N][N], int row, int col, int num) {
            for (int x = 0; x < N; x++) {
                if (grid[row][x] == num || grid[x][col] == num) return 0;
            }
        
            int startRow = row - row % 3, startCol = col - col % 3;
            for (int i = 0; i < 3; i++) {
                for (int j = 0; j < 3; j++) {
                    if (grid[i + startRow][j + startCol] == num) return 0;
                }
            }
        
            return 1;
        }
        
        int solveSudoku(int grid[N][N]) {
            for (int row = 0; row < N; row++) {
                for (int col = 0; col < N; col++) {
                    if (grid[row][col] == 0) {
                        for (int num = 1; num <= 9; num++) {
                            if (isSafe(grid, row, col, num)) {
                                grid[row][col] = num;
                                if (solveSudoku(grid)) return 1;
                                grid[row][col] = 0;
                            }
                        }
                        return 0;
                    }
                }
            }
            return 1;
        }

Music Player

Play WAV audio files in a console-based application.

Tip: Use libraries like SDL2 for handling audio playback in a simple way.

        #include <SDL2/SDL.h>
        #include <stdio.h>
        
        void playMusic(const char *filePath) {
            if (SDL_Init(SDL_INIT_AUDIO) < 0) {
                printf("Error initializing SDL: %s\n", SDL_GetError());
                return;
            }
        
            if (SDL_LoadWAV(filePath, &spec, &audioBuffer, &audioLength) == NULL) {
                printf("Error loading WAV file: %s\n", SDL_GetError());
                SDL_Quit();
                return;
            }
        
            SDL_AudioDeviceID device = SDL_OpenAudioDevice(NULL, 0, &spec, NULL, 0);
            SDL_QueueAudio(device, audioBuffer, audioLength);
            SDL_PauseAudioDevice(device, 0);
        
            SDL_Delay((audioLength / spec.freq) * 1000); // Wait for the audio to finish
        
            SDL_CloseAudioDevice(device);
            SDL_FreeWAV(audioBuffer);
            SDL_Quit();
        }
        
        int main() {
            playMusic("example.wav");
            return 0;
        }
